\section{Executive Summary}

\begin{tcolorbox}[projectbox]
    \textbf{Project Overview:} An interactive visual novel that combines professional photography, AI-generated imagery, and branching narrative to create an immersive storytelling experience centered around artistic collaboration and personal growth.
\end{tcolorbox}

\vspace{0.5cm}

\textbf{Zara: Light \& Shadow} is a production-quality visual novel that tells the story of an aspiring model navigating the world of professional photography, artistic expression, and personal boundaries. The project showcases:

\begin{itemize}[leftmargin=*, itemsep=8pt]
    \item \textbf{Real Artist as Protagonist:} Authentic character development based on professional modeling shoots
    \item \textbf{Professional Photography Integration:} 10+ thematic photographs capturing different moods and narrative moments
    \item \textbf{Cinematic Transitions:} 2-3 short video clips (15-30 seconds) connecting chapter progressions
    \item \textbf{Interactive Storytelling:} Player choices affect character relationships and story outcomes
    \item \textbf{Multiple Endings:} Branching narrative paths based on trust, comfort, and artistic vision
\end{itemize}

\vspace{0.5cm}

\subsection*{Current Development Stage}

\begin{tabular}{ll}
    \textbf{Status:} & Chapter 1 - Early Development \\
    \textbf{Character Shoots:} & Test Photography Completed \\
    \textbf{Technical Platform:} & React/TypeScript Web Application \\
    \textbf{Target Audience:} & Adults 18+ interested in narrative games \\
    \textbf{Release Timeline:} & Pre-release version available for review \\
\end{tabular}

\section{Project Structure}

\subsection{Chapter 1: Light \& Shadow}

\begin{tcolorbox}[colback=secondary!5, colframe=secondary]
    \textbf{Setting:} Professional photography studio over a single day\\
    \textbf{Duration:} 45-60 minutes of gameplay\\
    \textbf{Scenes:} 10 interconnected narrative beats\\
    \textbf{Choices:} 15+ player decisions affecting outcomes
\end{tcolorbox}

\subsubsection*{Scene Progression}

\begin{enumerate}[leftmargin=*, itemsep=8pt]
    \item \textbf{Morning Arrival} - First impressions and establishing boundaries
    \item \textbf{Wardrobe Discussion} - Building rapport and understanding vision
    \item \textbf{Light Test \& Preparation} - Technical professionalism meets personal connection
    \item \textbf{First Frames} - Capturing authentic emotion
    \item \textbf{Creative Exploration} - Player choices unlock different artistic directions
    \item \textbf{Equipment Hiccup} - Handling challenges builds trust
    \item \textbf{Intimate Portraiture} - Vulnerable moments (consent-gated)
    \item \textbf{Artistic Experimentation} - Pushing creative boundaries together
    \item \textbf{Climactic Session} - Multiple path-dependent scenarios
    \item \textbf{Reflection \& Future} - Endings based on relationship variables
\end{enumerate}

\subsection{Interactive Mechanics}

\begin{itemize}[leftmargin=*, itemsep=8pt]
    \item \textbf{Variable Tracking:} Comfort, Trust, Artistic Cohesion, Intimacy
    \item \textbf{Branching Paths:} 5+ distinct endings based on player choices
    \item \textbf{Consent System:} Explicit player confirmation for intimate content
    \item \textbf{Save/Load:} Standard VN features with text history
    \item \textbf{Auto-Advance:} Configurable reading pace
\end{itemize}
